4 Histograms

# Def data
dados = cars$speed
# Construindo um histograma -  usado para visualizar a distribuio de freq de UMA varivel.
# divide em classes (grupos) de dados que aparecem com maior/menor frequencia.
hist(dados)

# ?hist Conforme consta no help, o parmetro breaks pode ser um dos itens abaixo: Um vetor para os pontos de quebra entre as clulas do histograma Uma funo para calcular o vetor de breakpoints Um nico nmero que representa o nmero de clulas para o histograma Uma cadeia de caracteres que nomeia um algoritmo para calcular o nmero de clulas Uma funo para calcular o nmero de clulas.

hist(dados, breaks = 10, main = "Histograma das Velocidades") # breaks: caixinhas

hist(dados, labels = T, breaks = c(0,5,10,20,30), main = "Histograma das Velocidades") # caixinhas especficas

hist(dados, labels = T, breaks = 10, main = "Histograma das Velocidades") # mostra o total de freq das caixinhas

hist(dados, labels = T, ylim = c(0,10), breaks = 10, main = "Histograma das Velocidades")

# Adicionando linhas ao histograma
grafico <- hist(dados, breaks = 10, main = "Histograma das Velocidades")
# Criando uma distribuio normal, media, desvio padro 
xaxis = seq(min(dados), max(dados), length = 10)
yaxis = dnorm(xaxis, mean = mean(dados), sd = sd(dados))
#calcular a diferena:
yaxis = yaxis*diff(grafico$mids)*length(dados)

#acrescentar mais uma camada = distribuio normal 
lines(xaxis, yaxis, col = "red")

distribuio normal mostra se os dados esto com: media ~ = 0 desvio padro ~ = 1

5 Pie

# Criando as fatias
fatias = c(40, 20, 40)
# Nomeando os labels
paises = c("Brasil", "Argentina", "Chile")
# Unindo paises e fatias - com a f paste
paises = paste(paises, fatias)
# Incluindo mais detalhes no label
paises = paste(paises, "%", sep = "")
# Construindo um grfico
pie(fatias, labels = paises,
    col = c("darksalmon", "gainsboro", "lemonchiffon4"), 
    main ="Distribuio de Vendas")

# Def dataframe (iris)
attach(iris)
Values = table(Species)

# 3D - Pcte Plotrix
library(plotrix)
